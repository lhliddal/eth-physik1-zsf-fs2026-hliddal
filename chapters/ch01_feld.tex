\StartChapter{Elektrisches Feld}

\begin{runintext}
Formeln zu Kräften und Feldern von Punktladungen und Ladungsverteilungen: \ZSFkeyword{Coulomb-Kraft}, elektrisches Feld einzelner und vieler Ladungen, \ZSFkeyword{Dipole}, Feldlinien, Fluss und \ZSFkeyword{Gauss-Gesetz}. Nützlich bei Aufgaben mit Ladungen, Feldverläufen, Fluss und Symmetrie-Gauss-Flächen.
\end{runintext}

\SubsectionBar[sec:coulomb-charge]{Ladung, Coulomb}
\begin{runintext}
Formeln für Ladungsmengen und die Kraft zwischen zwei \ZSFkeyword{Punktladungen}; brauchst du, wenn in Aufgaben einzelne Ladungen und ihre Wechselwirkung im Abstand $r$ vorkommen.
\end{runintext}
\begin{defbox}[Elementarladung]
\ZSFkeyword{Quantisierung}: Ladung in Vielfachen von $e$. Elektron: $-e$, Proton: $+e$.
\end{defbox}
\begin{formulabox}
\[ e = 1{,}60 \times 10^{-19}\,\si{C} \]
\formulanote{Zweck: Betrag der Elementarladung.}
\end{formulabox}
\begin{runintext}
Ladung bleibt erhalten (weder erzeugt noch vernichtet). \ZSFkeyword{Leiter}: delokalisierte Elektronen; Nichtleiter: lokalisierte. \ZSFkeyword{Influenz}: Leiter erden, Ladung nähern, Erde trennen, Ladung entfernen.
\end{runintext}
\begin{defbox}[Coulomb'sches Gesetz]
Kraft von $q_1$ auf $q_2$ im Abstand $r$; $\hat{r}$ von $q_1$ nach $q_2$.
\end{defbox}
\begin{formulabox}
\[ \vec{F} = \frac{1}{4\pi\varepsilon_0}\,\frac{q_1 q_2}{r^2}\,\hat{r} \]
\formulanote{Zweck: Kraft zwischen zwei Punktladungen.}
\formulasep
\[ \frac{1}{4\pi\varepsilon_0} = 8{,}99\times 10^9\,\si{N.m^2/C^2} \]
\[ \varepsilon_0 = 8{,}85\times 10^{-12}\,\si{C^2/(N.m^2)} \]
\formulanote{Zweck: Coulomb-Konstante bzw.\ Einheit.}
\end{formulabox}

\SubsectionBar[sec:efield-superposition]{Elektrisches Feld, Superposition}
\begin{runintext}
Formeln für das elektrische Feld durch eine oder viele Ladungen; nützlich, wenn Feldstärken in Punkten und die Überlagerung mehrerer Ladungen gefragt sind.
\end{runintext}
\begin{formulabox}
\[ \vec{E} = \frac{\vec{F}}{q_0} \]
\formulanote{Zweck: Definition des elektrischen Feldes (Kraft $\vec{F}$ pro Probeladung $q_0$).}
\end{formulabox}

\begin{formulabox}
\[ \vec{E} = \frac{1}{4\pi\varepsilon_0}\,\frac{q}{r^2}\,\hat{r} \quad \text{Punktladung} \]
\formulanote{Zweck: Elektrisches Feld einer Punktladung.}
\end{formulabox}

\begin{formulabox}
\[ \vec{E} = \sum_i \vec{E}_i = \sum_i \frac{1}{4\pi\varepsilon_0}\,\frac{q_i}{r_i^2}\,\hat{r}_i \quad \text{Superposition} \]
\formulanote{Zweck: Superposition (elektrisches Feld mehrerer Punktladungen).}
\end{formulabox}
\begin{runintext}
\ZSFkeyword{Feldlinien}: von + nach $-$; Dichte $\propto$ Feldstärke.
\end{runintext}

\SubsectionBar[sec:dipole]{Dipol}
\begin{runintext}
Formeln für \ZSFkeyword{Dipolmoment}, \ZSFkeyword{Drehmoment} und Energie eines Dipols im Feld; brauchst du bei Aufgaben zu polaren Molekülen oder kleinen Ladungspaaren im homogenen \ZSFkeyword{elektrischen Feld}. \ZSFkeyword{Dipol}: zwei entgegengesetzt gleiche Ladungen, Abstand $\vec{\ell}$ (von $-$ nach $+$).
\end{runintext}

\begin{defbox}[Elektrischer Dipol \& Feldlinien]
\begin{splitbox}[0.65]
    \centering
    \includegraphics[width=\linewidth, keepaspectratio]{graphics/electric_dipole.png}
    \ZSFfigcaption{Zwei Ladungen $\pm q$ im Abstand $\vec{\ell}$}
    \tcblower
    \centering
    \includegraphics[width=\linewidth, keepaspectratio]{graphics/Elektrostatik_Dipol_Feldlinien_Symmetrie.png}
    \ZSFfigcaption{Feldlinien verlaufen von $+$ nach $-$}
\end{splitbox}
\end{defbox}

\begin{formulabox}
\[ \vec{p} = q\,\vec{\ell} \quad \text{Dipolmoment} \]
\formulanote{Zweck: Definition des elektrischen Dipolmoments.}
\formulasep
\[ \vec{M} = \vec{p} \times \vec{E} \quad \text{Drehmoment im homogenen elektrischen Feld} \]
\formulanote{Zweck: Drehmoment auf Dipol im elektrischen Feld.}
\formulasep
\[ E_{\mathrm{pot}} = -\vec{p}\cdot\vec{E} + U_0 \quad \text{($U_0$ oft $=0$)} \]
\formulanote{Zweck: Potenzielle Energie des Dipols im elektrischen Feld.}
\formulasep
\[ \vec{E}_{\mathrm{axial}} \approx \frac{1}{4\pi\varepsilon_0}\frac{2\vec{p}}{r^3}, \quad \vec{E}_{\mathrm{quer}} \approx -\frac{1}{4\pi\varepsilon_0}\frac{\vec{p}}{r^3} \quad (r \gg \ell) \]
\formulanote{Zweck: Dipol-Fernfeld auf Achse (axial) und quer dazu (Mittelsenkrechte).}
\end{formulabox}
\begin{runintext}
\ZSFkeyword{Polare Moleküle}: permanentes $\vec{p}$; nichtpolar: induzierbar.
\end{runintext}

\SubsectionBar[sec:charge-distributions]{Ladungsverteilungen}
\begin{runintext}
Formel, um das \ZSFkeyword{elektrische Feld} \ZSFkeyword{kontinuierlicher Ladungsverteilungen} (Volumen-, Flächen- oder Linienladungen) über ein Integral zu berechnen; nützlich bei ausgedehnten Ladungswolken, Drähten oder Flächen.
\end{runintext}
\begin{formulabox}
\[ \vec{E} = \int \mathrm{d}\vec{E} = \frac{1}{4\pi\varepsilon_0} \int \frac{\mathrm{d}q}{r^2}\,\hat{r} \]
\formulanote{Zweck: elektrisches Feld durch integration über Ladungsverteilung.}
\end{formulabox}
\begin{runintext}
$\mathrm{d}q = \rho\,\mathrm{d}V$ (Raum), $\mathrm{d}q = \sigma\,\mathrm{d}A$ (Fläche), $\mathrm{d}q = \lambda\,\mathrm{d}\ell$ (Linie).
\end{runintext}
\begin{procedure}[Rezept zur Herleitung des elektrischen Feldes]
  \ProcStep{Ladungselement aufstellen}{%
    \[ \mathrm{d}q = \lambda\,\mathrm{d}x \quad / \quad \sigma\,\mathrm{d}A \quad / \quad \rho\,\mathrm{d}V \]
  }
  \ProcStep{Symmetrien nutzen \& Komponenten zerlegen}{%
    Nur relevante Feldrichtung betrachten.
    \[ \mathrm{d}E_x = \mathrm{d}E \cos\theta = \frac{1}{4\pi\varepsilon_0}\,\frac{\mathrm{d}q}{r^2}\cos\theta \]
    \formulanote{Tipp: Drücke $r$, $x$ und $\cos\theta$ geometrisch aus (oft über Winkel $\vartheta$, siehe Winkelparametrisierung \ZSFref{sec:1-winkel}).}
  }
  \ProcStep{Aufintegrieren}{%
    Grenzen bestimmen und integrieren.
    \[ E_x = \int \mathrm{d}E_x \]
  }
\end{procedure}

\SubsectionBar[sec:1-winkel]{Winkelparametrisierung}
\begin{runintext}
\ZSFkeyword{Winkelparametrisierung}: Geniales Rezept für Integrale über Linienleiter ($\vec{E}, \vec{B}$). Wandelt die unschöne Integration über den Ort $x$ in eine simple trigonometrische Integration über den Winkel $\vartheta$ um.
\end{runintext}
\begin{procedure}[Winkelparametrisierung]
  \ProcStep{Substitution}{%
    $h$ = Lotabstand, $\vartheta$ = Winkel relativ zum Lot.
    \[ \begin{aligned}
        x &= h \cdot \tan\vartheta \quad\implies\quad \mathrm{d}x = \frac{h}{\cos^2\vartheta}\,\mathrm{d}\vartheta \\[1ex]
        r &= \frac{h}{\cos\vartheta} \quad\implies\quad \frac{\mathrm{d}x}{r^2} = \frac{1}{h}\,\mathrm{d}\vartheta
    \end{aligned} \]
  }
  \ProcStep{Elektrisches Feld einer Linienladung $\lambda$}{%
    \[ \begin{aligned}
        E_{\parallel} &= \frac{\lambda}{4\pi\varepsilon_0 h} \int_{\vartheta_1}^{\vartheta_2} \sin\vartheta\,\mathrm{d}\vartheta = \frac{\lambda}{4\pi\varepsilon_0 h} \Bigl[-\cos\vartheta\Bigr]_{\vartheta_1}^{\vartheta_2} \\[2ex]
        E_{\perp} &= \frac{\lambda}{4\pi\varepsilon_0 h} \int_{\vartheta_1}^{\vartheta_2} \cos\vartheta\,\mathrm{d}\vartheta = \frac{\lambda}{4\pi\varepsilon_0 h} \Bigl[\sin\vartheta\Bigr]_{\vartheta_1}^{\vartheta_2}
    \end{aligned} \]
  }
  \ProcStep{B-Feld eines geraden Stromleiters $I$}{%
    \[
        B = \frac{\mu_0 I}{4\pi h} \int_{\vartheta_1}^{\vartheta_2} \cos\vartheta\,\mathrm{d}\vartheta = \frac{\mu_0 I}{4\pi h} \Bigl[\sin\vartheta\Bigr]_{\vartheta_1}^{\vartheta_2}
    \]
    \formulanote{Unendlicher Leiter ($\vartheta \in [-\pi/2, \pi/2]$): $E_{\perp} = \frac{\lambda}{2\pi\varepsilon_0 h}$, $B = \frac{\mu_0 I}{2\pi h}$. Komponente $E_{\parallel}$ hebt sich auf.}
  }
\end{procedure}

\SubsectionBar[sec:electric-flux]{Elektrischer Fluss}
\begin{runintext}
Formel für den \ZSFkeyword{elektrischen Fluss} durch eine Fläche; brauchst du, wenn in Aufgaben Feldlinienanzahl oder Fluss durch eine gegebene Fläche vorkommt und als Vorbereitung für das \ZSFkeyword{Gauss-Gesetz}.
\end{runintext}
\begin{formulabox}
\[ \Phi_{\mathrm{el}} = \lim_{\Delta A_i\to 0} \sum_i \vec{E}_i\cdot\Delta\vec{A}_i = \int_A \vec{E}\cdot\mathrm{d}\vec{A} \]
\formulanote{Zweck: Elektrischer Fluss durch Fläche $A$ (anschaulich: Nettozahl der Feldlinien durch $A$).}
\end{formulabox}

\SubsectionBar[sec:gauss-law]{Gauss}
\begin{runintext}
Integralform des \ZSFkeyword{Gauss-Gesetzes} und Randwerte am Leiter; nützlich zur Bestimmung von \ZSFkeyword{elektrischen Feldern} stark symmetrischer Ladungsverteilungen ohne explizite Integration.
\end{runintext}
\begin{formulabox}
\[ \Phi_{\mathrm{el}} = \oint_A \vec{E}\cdot\mathrm{d}\vec{A} = \oint_A E_n\,\mathrm{d}A = \frac{q_{\mathrm{innen}}}{\varepsilon_0} \]
\formulanote{Zweck: Gauss'sches Gesetz; Fluss $\propto$ eingeschlossene Ladung.}
\end{formulabox}

\begin{figbox}[Visualisierung Gauss'sches Gesetz]
\centering
\includegraphics[width=\linewidth, keepaspectratio]{graphics/gauss_flux_explanation.png}
\ZSFfigcaption{Netto-Fluss $\Phi_{\mathrm{el}}$ (austretende Feldlinien) hängt nur von eingeschlossener Netto-Ladung ab.}
\end{figbox}
\ZSFgapS

\begin{formulabox}
\[
\begin{gathered}
\oint_A \vec{E}\cdot\mathrm{d}\vec{A} = E(r)\,4\pi r^2 = \frac{q_{\mathrm{innen}}(r)}{\varepsilon_0} \\[-2pt]
\text{Kugelsymmetrie (auf Kugelflaeche ist $E$ konstant)}
\end{gathered}
\]
\[ E(r) = \frac{1}{4\pi\varepsilon_0}\,\frac{q_{\mathrm{innen}}(r)}{r^2} \]
\formulanote{Zweck: Vereinfachte Gauss-Form bei Kugelsymmetrie; kein neues Gesetz.}
\end{formulabox}
\begin{runintext}
\ZSFkeyword{Leiter}: Gesamte Ladung sitzt auf der Oberfläche; im Inneren gilt $E=0$ und $q=0$.
\end{runintext}
\begin{formulabox}
\[ \Delta E_n = E_{n,+} - E_{n,-} = \frac{\sigma}{\varepsilon_0} \quad \text{Unstetigkeit} \]
\[ E_n = \frac{\sigma}{\varepsilon_0} \quad \text{unmittelbar außerhalb Leiter} \]
\formulanote{Zweck: Feld-Sprung ($\Delta E_n$) beim Durchqueren einer geladenen Fläche. Am Leiter gilt $E_{n,-}=0$ (innen) $\implies E_n = \frac{\sigma}{\varepsilon_0}$ (aussen).}
\end{formulabox}

\SubsectionBar[sec:symmetric-fields]{Symmetrische Felder}
\begin{runintext}
Fertige Resultate für \ZSFkeyword{elektrische Felder} typischer Symmetrien (Linienladung, Ring, Scheibe, Ebene, Kugel); praktisch zum schnellen Einsetzen in Zahlenaufgaben mit diesen \ZSFkeyword{Standardgeometrien}.
\end{runintext}
\begin{formulabox}
\[
\begin{gathered}
\vec{E} = \frac{1}{2\pi\varepsilon_0}\,\frac{\lambda}{r_\perp}\,\hat{r}_\perp \\[-2pt]
\text{Linienladung unendlich}
\end{gathered}
\]
\end{formulabox}

\begin{formulabox}
\[ \lambda = \sigma\,2\pi r \quad \text{zylindrischer Leiter} \]
\formulanote{Zweck: Umrechnung zwischen Flaechenladungsdichte $\sigma$ und Linienladungsdichte $\lambda$ bei Radius $r$.}
\end{formulabox}

\begin{formulabox}
\[
\begin{gathered}
E_z = \frac{1}{4\pi\varepsilon_0}\,\frac{qz}{(z^2+a^2)^{3/2}} \\[-2pt]
\text{Ringachse}
\end{gathered}
\]
\end{formulabox}

\begin{formulabox}
\[
\begin{gathered}
E_z = \frac{\sigma}{2\varepsilon_0}\,\sgn(z)\left(1 - \left(1+\frac{r_S^2}{z^2}\right)^{-1/2}\right) \\[-2pt]
\text{Kreisscheibe Achse}
\end{gathered}
\]
\end{formulabox}

\begin{formulabox}
\[
\begin{gathered}
E = E_z = \sgn(z)\,\frac{\sigma}{2\varepsilon_0} \\[-2pt]
\text{Unendliche Ebene}
\end{gathered}
\]
\end{formulabox}

\begin{formulabox}
\[
\begin{gathered}
E = \dfrac{1}{4\pi\varepsilon_0}\,\dfrac{q}{r^2}\;\text{für $r>r_K$};\quad E=0\;\text{für $r<r_K$} \\[-2pt]
\text{Kugelschale}
\end{gathered}
\]
\formulanote{Zweck: elektrisches Feld typischer symmetrischer Verteilungen (Linie, Ring, Scheibe, Ebene, Kugel).}
\end{formulabox}

\begin{formulabox}
\[
\begin{gathered}
\vec{E}(r) = \frac{\rho}{3\varepsilon_0}\,\vec{r} \qquad (r<R) \\[-2pt]
\text{homogen geladene Vollkugel (innen)}
\end{gathered}
\]
\formulanote{Aus Gau\ss{}: nur die eingeschlossene Ladung traegt bei, daher waechst $E$ im Inneren linear mit $r$.}
\end{formulabox}

\ZSFgapS
\begin{procedure}[Ortsabhängige Ladungsdichte $\rho(r)$]
  \ProcStep{Eingeschlossene Ladung für $r<R$}{%
    \ZSFdanger{Achtung:} Falls $\rho = \rho(r)$, zuerst die eingeschlossene Ladung bis zum aktuellen Radius bestimmen.
    \[ q_{\mathrm{in}}(r) = 4\pi \int_0^r \rho(r') (r')^2\,\mathrm{d}r' \]
  }
  \ProcStep{Elektrisches Feld innen bestimmen}{%
    \[ E_{\mathrm{innen}}(r) = \frac{q_{\mathrm{in}}(r)}{4\pi\varepsilon_0 r^2} = \frac{1}{\varepsilon_0 r^2}\int_0^r \rho(r') (r')^2\,\mathrm{d}r' \]
    \formulanote{Aussen ($r > R$): Das Feld verhält sich wie eine Punktladung mit der Gesamtladung $Q = q_{\mathrm{in}}(R)$ im Ursprung.}
  }
\end{procedure}
