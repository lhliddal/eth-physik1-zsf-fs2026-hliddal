\StartChapter{Elektromagnetische Induktion}

\begin{runintext}
Zusammenhang zwischen zeitlich veränderlichen Magnetfeldern und induzierten elektrischen Spannungen. Kernthemas sind der \ZSFkeyword[Magnetischer Fluss]{magnetische Fluss}, das \ZSFkeyword[Faraday'sches Gesetz]{Faraday'sche Gesetz}, die \ZSFkeyword{Lenz'sche Regel} sowie induktive Bauteile und deren Energiedichte.
\end{runintext}

\SubsectionBar[sec:magnetic-flux]{Magnetischer Fluss}
\begin{runintext}
Das "Zählen" der Magnetfeldlinien durch eine vorgegebene Querschnittsfläche; die entscheidende Grosse zur Berechnung von \ZSFkeyword*{Induktionsphänomenen}.
\end{runintext}
\begin{formulabox}
\[ \Phi_{\mathrm{mag}} = \int_A \vec{B}\cdot\mathrm{d}\vec{A} \quad \text{pro Windung} \]
\formulanote{Zweck: $\Phi_{\mathrm{mag}}$ = Fluss durch \emph{eine} Leiterschleife (Fläche $A$).}
\formulasep
\[ 1\,\si{Wb} = 1\,\si{T.m^2} \]
\[ \Phi_{\mathrm{mag}} = |\vec{B}|\,|A|\,\cos\theta \quad \text{homogen} \]
\formulasep
\[ n\,\Phi_{\mathrm{mag}} = L\,I \quad \text{Spule mit $n$ Wind.} \]
\[ n_1\Phi_{\mathrm{mag},1} = L_1 I_1 + L_{12} I_2,\quad n_2\Phi_{\mathrm{mag},2} = L_2 I_2 + L_{21} I_1 \]
\formulanote{Konvention: $\Phi_{\mathrm{mag}}$ = Fluss \emph{pro Windung}; Gesamtfluss einer Spule mit $n$ Wind.\ ist $n\,\Phi_{\mathrm{mag}}$.}
\end{formulabox}

\SubsectionBar[sec:faradays-law]{Faraday}
\begin{runintext}
Das \ZSFkeyword{Induktionsgesetz}: Jede Änderung des \ZSFkeyword[Magnetischer Fluss]{magnetischen Flusses} ruft eine \ZSFkeyword{Ringspannung} (und damit ein \ZSFkeyword[Elektrisches Feld]{elektrisches Feld}) hervor.
\end{runintext}

\ZSFfig{graphics/Induktion_Leiterschleife_Flaechenaenderung_Delta_A_v.png}{{Bewegter Stab, Flächenänderung}}{$\Delta A=\ell\,v\,\Delta t$ liefert $U_{\text{ind}}=-B\ell v$}

\begin{formulabox}
\[ U_{\mathrm{ind}} = -\frac{\mathrm{d}\Phi_{\mathrm{mag}}}{\mathrm{d}t} \quad \text{Faraday, eine Windung} \]
\[ U_{\mathrm{ind}} = -n\,\frac{\mathrm{d}\Phi_{\mathrm{mag}}}{\mathrm{d}t} \quad \text{Spule mit $n$ Wind.} \]
\formulanote{Zweck: Induzierte Spannung bei Flussänderung ($U_{\mathrm{ind}, \mathrm{mittel}} = -n\,\frac{\Delta\Phi_{\mathrm{mag}}}{\Delta t}$ für zeitliche Mittelwerte; Minuszeichen gemäß Lenz).}
\end{formulabox}

\begin{formulabox}
\[ U_{\mathrm{ind}} = \oint_C \vec{E}\cdot\mathrm{d}\vec{\ell} \quad \text{zeitabhängiges B} \]
\formulanote{Zweck: Induzierte Spannung bei zeitabhängigem Magnetfeld (Umlauf-Integral des elektrischen Wirbelfelds).}
\end{formulabox}

\begin{formulabox}
\[ U_{\mathrm{ind}} = |\vec{v}|\,|\vec{B}|\,\ell \quad \text{bewegter Stab $\perp$ zu $\vec{B}$} \]
\formulanote{Zweck: Bewegungsinduktion (Lorentzkraft auf Ladungsträger im bewegten Leiter).}
\end{formulabox}

\begin{formulabox}
\[ U_{\mathrm{ind}} = -L\,\frac{\mathrm{d}I}{\mathrm{d}t} \quad \text{Selbstinduktion} \]
\formulanote{Zweck: Induzierte Gegenspannung durch zeitliche Änderung des eigenen Stroms (Selbstinduktion).}
\end{formulabox}

\SubsectionBar[sec:lenz-law]{Lenz}
\begin{runintext}
Wichtige Regel, um aus der \ZSFkeyword{Energieerhaltung} schnell die Richtung des induzierten Stromes zu bestimmen.
\end{runintext}
\begin{defbox}[Lenz'sche Regel]
Induktionsspannung und -strom wirken ihrer \ZSFkeyword*{Ursache entgegen}. Oder: Der durch Flussänderung induzierte Strom erzeugt einen Fluss, der der Änderung entgegenwirkt.
\end{defbox}

\SubsectionBar[sec:self-mutual-inductance]{Selbst-, Gegeninduktion}
\begin{runintext}
Die \ZSFkeyword{Induktivität} beschreibt die Eigenschaft von Spulen und Leitern, den Stromfluss aufgrund des selbstaufgebauten oder externen Magnetfeldes zu trägen (\ZSFkeyword[Selbstinduktion]{Selbst-} und \ZSFkeyword{Gegeninduktion}).
\end{runintext}
\begin{formulabox}
\[ L = \frac{n\,\Phi_{\mathrm{mag}}}{I} \]
\formulanote{Selbstinduktivität (Def., Spule mit $n$ Wind., $[L] = \si{H}$).}
\[ L = \mu_0\,\frac{n^2\,A}{\ell} \quad \text{Selbstind.\ Zylinderspule} \]
\formulanote{Zweck: Definition der Selbstinduktivität $L$ und deren Geometrieformel für eine lange Zylinderspule.}
\end{formulabox}

\begin{formulabox}
\[ M = L_{\underbrace{\scriptstyle 12}_{\text{\tiny in 1 von 2}}} = L_{\underbrace{\scriptstyle 21}_{\text{\tiny in 2 von 1}}} = \frac{\overbrace{n_2}^{\text{\tiny Wind.\ 2}}\,\overbrace{\Phi_{\mathrm{mag},21}}^{\text{\tiny Fluss in 2 von 1}}}{\underbrace{I_1}_{\text{\tiny Strom 1}}} \]
\formulanote{Def.\ Gegeninduktivität ($[M] = \si{H}$), symmetrisch: $M_{12} = M_{21} = M$ (Reihenfolge egal). Indexkonvention: $\Phi_{\mathrm{mag},ab}$ = Fluss in Spule~$a$ (Ort) durch Strom $I_b$ in Spule~$b$ (Quelle).}
\[ U_{\mathrm{ind},2} = -M\,\frac{\mathrm{d}I_1}{\mathrm{d}t} \]
\formulanote{Induzierte Spannung in Spule 2 durch Stromänderung in Spule 1.}
\end{formulabox}

\begin{formulabox}
\[ M \approx \mu_0\,\frac{n_1\,n_2\,A_{\mathrm{innen}}}{\ell_{\mathrm{aussen}}} \quad \text{koaxiale lange Spulen} \]
\formulanote{Zweck: Geometrieformel für $M$ bei eng gekoppelten Spulen ($A_{\mathrm{innen}}$: Querschnitt der inneren Spule, $\ell_{\mathrm{aussen}}$: Länge der äußeren Spule).}
\[ 1\,\si{H} = 1\,\si{V.s/A} \]
\[ \mu_0 = 4\pi\times 10^{-7}\,\si{H/m} \]
\formulanote{Zweck: Einheiten und Naturkonstanten für Induktivitäten; $L$ = Selbstinduktivität (eine Spule); $M$ = Gegeninduktivität (zwei gekoppelte Spulen).}
\end{formulabox}

\SubsectionBar[sec:rl-circuits]{RL-Schaltungen}
\ZSFindex{RL-Kreis}
\begin{runintext}
Zeitabhängiges Anwachsen und Abklingen von Strömen in einem Schaltkreis mit \ZSFkeyword{Spule} und Serienwiderstand.
\end{runintext}
\begin{formulabox}
\[ U_0 - L\,\frac{\mathrm{d}I}{\mathrm{d}t} - I R = 0 \quad \text{Maschengl.\ (Laden)} \]
\formulasep
\[ I(t) = \frac{U_0}{R}\,(1 - \mathrm{e}^{-t/\tau}) = I_E\,(1 - \mathrm{e}^{-t/\tau}) \quad \text{Einschalten} \]
\[ \tau = L/R \quad \text{Zeitkonstante} \]
\formulasep
\[ I(t) = I_0\,\mathrm{e}^{-t/\tau} \quad \text{Ausschalten / Abbau} \]
\formulanote{Zweck: RL-Kreis. \ZSFdanger{Achtung:} Beim Einschalten wirkt die induzierte Spannung der Quelle entgegen (Lenz). Der Strom steigt träge mit Zeitkonstante $\tau = L/R$ an.}
\end{formulabox}

\SubsectionBar[sec:magnetic-field-energy]{Energie Magnetfeld}
\ZSFindex{Magnetische Feldenergie}
\begin{runintext}
Die im Magnetfeld und besonders in \ZSFkeyword[Induktivität]{Induktivitäten} (wie Spulen) zwischengespeicherte Energie, welche nach Abbau den Strom noch fliessen lässt.
\end{runintext}
\begin{formulabox}
\[ E_{\mathrm{mag}} = \frac{1}{2}\,L\,I^2 \]
\formulasep
\[ w_{\mathrm{mag}} = \frac{B^2}{2\mu_0} \quad \text{Energiedichte (für jedes $B$-Feld)} \]
\[ E_{\mathrm{mag}} = w_{\mathrm{mag}}\,V \quad \text{(homogenes Feld)} \]
\formulanote{Zweck: Gespeicherte magnetische Energie; Energiedichte im B-Feld; Gesamtenergie bei homogenem Feld über Volumen $V$.}
\end{formulabox}

\SubsectionBar[sec:eddy-currents]{Wirbelströme}
\ZSFindex{Wirbelstrom}
\begin{runintext}
Induzierte kreisförmige Ströme in massiven Leitern bei einer Änderung des \ZSFkeyword[Magnetischer Fluss]{magnetischen Flusses}.
\end{runintext}
\begin{statementbox}[Effekte \& Gegenmaßnahmen]
\ZSFkeyword*{Energieverlust}: Die Ströme erzeugen Joule'sche Wärme. Gegenmaßnahme in Transformatoren: \ZSFkeyword*{lamellierte} (geblechte) Eisenkerne. \ZSFgapXS
\ZSFkeyword*{Bremswirkung}: Induzierte Ströme erzeugen ein Gegenfeld, das der Bewegung entgegenwirkt (Lenz'sche Regel). Anwendung: \ZSFkeyword{Wirbelstrombremse}.
\end{statementbox}
